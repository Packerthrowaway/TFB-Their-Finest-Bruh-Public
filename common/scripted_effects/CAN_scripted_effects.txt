CAN_lower_conscription_crisis = { 
	if = {
		limit = {
			has_dynamic_modifier = {
				modifier = CAN_conscription_crisis_modifier
			}
		}
		custom_effect_tooltip = CAN_lower_conscription_crisis_tt_1
		hidden_effect = {
			remove_dynamic_modifier = { modifier = CAN_conscription_crisis_modifier }
			add_dynamic_modifier = { modifier = CAN_conscription_crisis_modifier_2 }
		}
	}
	else_if = {
		limit = {
			has_dynamic_modifier = {
				modifier = CAN_conscription_crisis_modifier_2
			}
		}
		custom_effect_tooltip = CAN_lower_conscription_crisis_tt_2
		hidden_effect = {
			remove_dynamic_modifier = { modifier = CAN_conscription_crisis_modifier_2 }
			add_dynamic_modifier = { modifier = CAN_conscription_crisis_modifier_3 }
		}
	}
	else = {
		remove_dynamic_modifier = { modifier = CAN_conscription_crisis_modifier_3 }
		clr_state_flag = CAN_conscription_crisis_flag
	}
}
